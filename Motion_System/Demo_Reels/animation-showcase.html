Implementation Examples

// Initialize CastMatch Motion System
const motion = new CastMatchMotion();

// Fade in animation
motion.fadeIn(element, {
  duration: 500,
  delay: 100,
  easing: 'easeOut'
});

// Cinematic reveal
motion.cinematicReveal(element, {
  scale: 0.8,
  blur: 10
});
